/* 
 * The purpose of this document is to aid in the
 * designing of the dirpl programming language.
 * This grammar file isn't very useful if you're
 * trying to learn the syntax of dirpl.  It does
 * a poor job of expressing the 2-dimensional aspect
 * of the language.
 */

program 
  : /* empty */
  | expr program
  | func_def program
  ;

expr
  : ID
  | NUM
  | func_call
  | expr '<' expr
  | expr '>' expr
  | expr '<=' expr
  | expr '>=' expr
  | expr '==' expr
  | expr '!=' expr
  | expr '|' expr
  | expr '&' expr
  | expr '-' expr
  | expr '+' expr
  | expr '*' expr
  | expr '\' expr
  | expr '\\' expr
  | expr '%' expr
  | 'ifelse()' expr expr expr
  ;

func_def
  : ':' ID formals
  ;

formals
  : '_'ID formals;
  : /* empty */
  ;

func_call
  : ID'()' args
  ;

args
  : ID args
  | INT args
  ;
