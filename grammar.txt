program : /* empty */
  | expr
  | func_def
  | func_call
  ;

expr
  : ID
  | NUM
  | expr '<' expr
  | expr '>' expr
  | expr '==' expr
  | expr '!=' expr
  | expr '|' expr
  | expr '&' expr
  | expr '-' expr
  | expr '+' expr
  | expr '*' expr
  | expr '\' expr
  ;

func_def
  : ':' ID formals
  ;

formals
  : '_' ID formals;
  : /* empty */
  ;

func_call
  : ID args
  ;

args
  : ID args
  | INT args
  ;
